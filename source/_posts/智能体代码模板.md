```java

WorkFlow supFlow = condFlow()
  .execute(ExistsWork.of("","Something went wrong.",10f))
  .when(WorkReportPredicate.COMPLETED)
  .then(seqFlow().execute(SaveWork.ofFail(ai,ac,""))
        .then(BIZLogWork.of("twitter平台注册失败"))
        .then(LogWork.ofFail(ai,ac,"发送的手机号或者IP被污染"))
        .build())
  .otherwise(PageTransitionWork.of("TW_signin_unlock_phone_postsend", "Enter your phone number", "confirmation code"))
  .build();
```

```java
 WorkFlow doubleCheck = seqFlow()
```
